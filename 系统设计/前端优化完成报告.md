### 2. 响应式布局
```typescript
// components/PageContainer.tsx
{
  display: 'flex',
  justifyContent: 'space-between',
  flexWrap: 'wrap',  // 关键：自动换行

}
```

### 3. 动态时间
```typescript
// MainLayout.tsx
useEffect(() => {

}, [])
```

## 🎓 最佳实践

### 1. 使用PageContainer
```tsx
<PageContainer
  title="页面标题"
  subtitle="英文副标题"
  extra={<Button>操作</Button>}
>
  <Card>内容</Card>
</PageContainer>
```

### 2. 统一按钮尺寸
```tsx
// 页面级按钮
<Button size="large">主要操作</Button>

// 表格内按钮
<Button size="small" type="link">编辑</Button>
```

### 3. 使用主题变量
```tsx
import { mintTheme } from '../theme/colors'

<Card style={{
  ...mintTheme.glass,
  borderRadius: mintTheme.borderRadius.xl
}} />
```
