## 1. 架构设计原则

### 1.1 场景约束

| 约束条件 | 说明 |
|----------|------|
| 运行环境 | 本地Mac电脑，非云服务商 |
| 用户规模 | 单管理员用户 |
| 数据规模 | 小规模数据，无需考虑大数据流量 |
| 部署方式 | 本地开发、编译、运行 |
| 网络环境 | 本地局域网访问 |

### 1.2 设计目标

| 设计目标 | 描述 |
|----------|------|
| **简洁实用** | 去掉企业级复杂架构，采用简单直接的技术方案 |
| **前后端分离** | 前后端解耦，独立开发维护 |
| **数据安全** | 定期备份，保证关机重启后数据一致性 |
| **快速开发** | 选择成熟稳定的技术栈，提高开发效率 |

### 1.3 设计原则

```
┌─────────────────────────────────────────────────────────────┐
│                        核心设计原则                          │
├─────────────────────────────────────────────────────────────┤
│  1. 前后端分离：后端Node.js API，前端React SPA              │
│  2. 简化认证：Basic Auth，单用户场景足够                    │
│  3. 简化数据：SQLite单文件数据库，备份简单                   │
│  4. 本地部署：直接运行，无需复杂容器化                       │
│  5. 简化监控：日志文件记录即可                               │
└─────────────────────────────────────────────────────────────┘
```

## 2. 技术选型

### 2.1 技术栈总览

| 层级 | 技术选型 | 说明 |
|------|----------|------|
| **前端** | React 18 + TypeScript | 成熟生态，组件化开发，类型安全 |
| **前端构建** | Vite | 快速开发服务器，冷启动快 |
| **前端UI** | Ant Design | 企业级UI组件库 |
| **状态管理** | Zustand | 轻量级状态管理 |
| **图表** | ECharts | 数据可视化 |
| **后端** | Node.js 20 + Express | JavaScript全栈，统一语言 |
| **数据库** | SQLite + better-sqlite3 | 高性能SQLite驱动 |
| **ORM** | Prisma | 现代化的ORM框架 |
| **认证** | Basic Auth | 简单用户名密码验证 |
| **任务调度** | node-cron | 定时任务（备份等） |

### 2.2 技术选型理由

| 技术 | 选型理由 |
|------|----------|
| **React 18** | 生态成熟，Hooks完善，社区活跃 |
| **TypeScript** | 类型安全，IDE支持好，降低运行时错误 |
| **Vite** | 开发体验好，冷启动快，热更新优秀 |
| **Ant Design** | 企业级UI组件库，组件丰富 |
| **Node.js + Express** | JavaScript全栈，统一技术栈，学习成本低 |
| **Prisma** | 类型安全，自动生成migration，IDE支持好 |
| **SQLite + better-sqlite3** | 单文件数据库，高性能，适合本地小规模应用 |

### 2.3 开发环境要求

| 环境 | 版本要求 |
|------|----------|
| 操作系统 | macOS 10.15+ |
| Node.js | 20.x LTS |
| npm | 10.x+ |
| 浏览器 | Safari / Chrome / Firefox 最新版 |

### 4.2 Prisma Schema

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = "file:../data/shop.db"
}

// 用户表
model User {
  id        Int      @id @default(autoincrement())
  username  String   @unique
  password  String
  realName  String?
  role      String  @default("admin")
  status    Int     @default(1)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

// 商品分类
model Category {
  id        Int      @id @default(autoincrement())
  parentId  Int      @default(0)
  name      String
  sort      Int      @default(0)
  status    Int      @default(1)
  createdAt DateTime @default(now())
  
  products  Product[]
}

// 商品
model Product {
  id            Int       @id @default(autoincrement())
  code          String    @unique
  name          String
  categoryId    Int
  brand         String?
  material      String?
  price         Float
  memberPrice   Float?
  costPrice     Float?
  mainImage     String?
  detailImages  String?   // JSON
  stockQuantity Int       @default(0)
  status        Int       @default(1)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  category      Category  @relation(fields: [categoryId], references: [id])
  skus          Sku[]
}

// SKU
model Sku {
  id            Int       @id @default(autoincrement())
  productId     Int
  skuCode       String    @unique
  specs         String?    // JSON: {"color": "红色"}
  price         Float
  stockQuantity Int       @default(0)
  status        Int       @default(1)
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  product       Product   @relation(fields: [productId], references: [id])
  inventories   Inventory[]
  orderItems    OrderItem[]
}

// 库存
model Inventory {
  id            Int       @id @default(autoincrement())
  skuId         Int
  warehouseId   Int       @default(1)
  quantity      Int       @default(0)
  safeQuantity  Int       @default(5)
  lastInTime    DateTime?
  lastOutTime   DateTime?
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  sku           Sku       @relation(fields: [skuId], references: [id])
  
  @@unique([skuId, warehouseId])
}

// 库存变动记录
model InventoryLog {
  id               Int       @id @default(autoincrement())
  skuId            Int
  changeType       String    // 'in', 'out', 'check', 'adjust'
  changeQuantity  Int
  beforeQuantity   Int
  afterQuantity    Int
  billType         String?
  billId           Int?
  operatorId       Int?
  remark           String?
  createdAt        DateTime  @default(now())
  
  sku              Sku       @relation(fields: [skuId], references: [id])
}

// 销售订单
model Order {
  id              Int       @id @default(autoincrement())
  orderNo         String    @unique
  memberId        Int?
  cashierId       Int
  totalQuantity   Int       @default(0)
  totalAmount     Float
  discountAmount  Float     @default(0)
  payAmount       Float
  payWay          String?    // JSON: {"cash": 500}
  pointsEarned    Int       @default(0)
  remark          String?
  status          Int       @default(1)
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  
  member          Member?   @relation(fields: [memberId], references: [id])
  cashier         User      @relation(fields: [cashierId], references: [id])
  items           OrderItem[]
  returns         OrderReturn[]
}

// 订单明细
model OrderItem {
  id            Int       @id @default(autoincrement())
  orderId       Int
  skuId         Int
  productName   String
  quantity      Int
  unitPrice     Float
  amount        Float
  createdAt     DateTime  @default(now())
  
  order         Order     @relation(fields: [orderId], references: [id])
  sku           Sku       @relation(fields: [skuId], references: [id])

// 退货单
model OrderReturn {
  id            Int       @id @default(autoincrement())

  orderId       Int
  returnReason  String?
  returnAmount  Float
  status        Int       @default(0)
  operatorId    Int?
  createdAt     DateTime  @default(now())
  
  order         Order     @relation(fields: [orderId], references: [id])

// 会员
model Member {
  id            Int       @id @default(autoincrement())

  name          String
  phone         String    @unique
```

### 4.3 索引设计

### 7.2 备份实现

```typescript
// backend/src/utils/backup.ts
import fs from 'fs';
import path from 'path';
import { execSync } from 'child_process';

const DATA_DIR = path.join(process.env.HOME!, 'Documents/薄荷集市/data');
const BACKUP_DIR = path.join(DATA_DIR, 'backup');

export async function backupDatabase(): Promise<string> {
  // 确保备份目录存在
  if (!fs.existsSync(BACKUP_DIR)) {
    fs.mkdirSync(BACKUP_DIR, { recursive: true });
  }
  
  const timestamp = new Date().toISOString().replace(/[-:]/g, '').split('.')[0];
  const backupFile = path.join(BACKUP_DIR, `shop_${timestamp}.db`);
  
  // 复制数据库文件
  const dbPath = path.join(DATA_DIR, 'shop.db');
  fs.copyFileSync(dbPath, backupFile);
  
  console.log(`数据库备份成功: ${backupFile}`);
  
  // 清理旧备份（保留30天）
  cleanupOldBackups(30);
  
  return backupFile;
}

function cleanupOldBackups(days: number): void {
  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - days);
  
  fs.readdirSync(BACKUP_DIR)
    .filter(f => f.startsWith('shop_') && f.endsWith('.db'))
    .forEach(filename => {
      const filepath = path.join(BACKUP_DIR, filename);
      const mtime = fs.statSync(filepath).mtime;
      if (mtime < cutoffDate) {
        fs.unlinkSync(filepath);
        console.log(`删除旧备份: ${filepath}`);
      }
    });
}
```

### 7.3 定时备份任务

```typescript
// backend/src/tasks/scheduler.ts
import cron from 'node-cron';
import { backupDatabase } from '../utils/backup';

export function initScheduler(): void {
  // 每天23:00执行备份
  cron.schedule('0 23 * * *', async () => {
    console.log('开始执行数据库备份...');
    try {
      await backupDatabase();
      console.log('数据库备份完成');
    } catch (error) {
      console.error('数据库备份失败:', error);
    }
  });
}
```

```typescript
// backend/src/utils/logger.ts
import fs from 'fs';
import path from 'path';

const LOG_DIR = path.join(process.env.HOME!, 'Documents/薄荷集市/logs');
if (!fs.existsSync(LOG_DIR)) {
  fs.mkdirSync(LOG_DIR, { recursive: true });
}

const logStream = fs.createWriteStream(
  path.join(LOG_DIR, 'app.log'),
  { flags: 'a' }
);

export const logger = {
  info: (msg: string) => {
    const log = `[${new Date().toISOString()}] INFO: ${msg}`;
```
