## 1. 项目概述

### 1.1 项目背景

薄荷集市是一家专注于时尚饰品销售的品牌店铺，主营产品包括戒指、项链、手链、耳饰、发饰、胸针等各类时尚配饰。随着业务规模不断扩大，传统的纸质记账和Excel表格管理已无法满足日常运营需求。

### 1.2 项目目标

本系统旨在为薄荷集市打造一套高效、稳定、易用的店铺管理系统：

- **业务数字化**：将商品管理、库存管理、销售管理、财务管理等核心业务环节全部纳入系统管理
- **流程自动化**：通过系统自动化处理日常业务，减少人工操作失误，提高工作效率
- **决策数据化**：提供多维度数据分析和可视化报表，为经营决策提供数据支撑

### 1.3 技术架构

| 层级 | 技术选型 |
|------|----------|
| 前端 | React 18 + TypeScript + Vite + Ant Design + Zustand + ECharts |
| 后端 | Node.js 20 + Express + TypeScript |
| 数据库 | SQLite + Prisma ORM |
| 认证 | Basic Auth (bcrypt密码) |

### 1.4 功能模块

| 模块 | 说明 | 优先级 |
|------|------|--------|
| 系统管理 | 用户管理、系统配置、操作日志 | P0 |
| 商品管理 | 商品档案、分类管理、品牌管理 | P0 |
| 库存管理 | 入库、出库、盘点、预警 | P0 |
| 销售管理 | POS收银、退换货、挂单取单 | P0 |
| 会员管理 | 会员档案、积分、等级 | P1 |
| 营销管理 | 促销、优惠券 | P1 |
| 财务报表 | 日结账、销售报表 | P1 |

### 2.2 安装步骤

```bash
# 使用 Homebrew 安装

node --version

```


```bash

# 安装依赖
```

### 4.2 数据表结构

#### 4.2.1 用户表 (User)

```typescript
// Prisma Schema
model User {
  id        Int      @id @default(autoincrement())
  username  String   @unique  // 用户名，唯一
  password  String            // bcrypt哈希后的密码
  realName  String?           // 真实姓名
  role      String  @default("admin")  // 角色
  status    Int     @default(1)         // 状态: 0-禁用, 1-启用
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  
  orders    Order[]  // 关联销售订单
}
```

#### 4.2.2 商品分类表 (Category)

```typescript
model Category {

  name      String                // 分类名称
  sort      Int     @default(0)   // 排序
  status    Int     @default(1)   // 状态
  createdAt DateTime @default(now())
  
  products  Product[]  // 关联商品

```

#### 4.2.3 商品表 (Product)

```typescript
model Product {
  id            Int       @id @default(autoincrement())
  code          String    @unique  // 商品编码
  name          String              // 商品名称
  categoryId    Int                 // 分类ID
  brand         String?             // 品牌
  material      String?             // 材质
  price         Float               // 吊牌价
  memberPrice   Float?              // 会员价
  costPrice     Float?              // 成本价
  mainImage     String?             // 主图URL
  detailImages  String?             // 详情图(JSON数组)
  stockQuantity Int     @default(0) // 库存数量
  status        Int     @default(1) // 状态: 0-下架, 1-上架
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  
  category      Category  @relation(fields: [categoryId], references: [id])
  skus          Sku[]
}
```

#### 4.2.4 SKU表 (Sku)

```typescript
model Sku {
  id            Int       @id @default(autoincrement())
  productId     Int                  // 商品ID
  skuCode       String    @unique   // SKU编码
  specs         String?              // 规格(JSON): {"color": "红色", "size": "M"}
  price         Float                // 价格
  stockQuantity Int     @default(0)  // 库存数量
  status        Int     @default(1) // 状态
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  
  product       Product   @relation(fields: [productId], references: [id])
  inventories   Inventory[]
  orderItems    OrderItem[]
}
```

#### 4.2.5 库存表 (Inventory)

```typescript
model Inventory {
  id            Int       @id @default(autoincrement())
  skuId         Int                  // SKU ID
  warehouseId   Int     @default(1) // 仓库ID
  quantity      Int     @default(0) // 库存数量
  safeQuantity  Int     @default(5) // 安全库存
  lastInTime    DateTime?           // 最后入库时间
  lastOutTime   DateTime?           // 最后出库时间
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt
  
  sku           Sku       @relation(fields: [skuId], references: [id])
  
  @@unique([skuId, warehouseId])  // 同一SKU同一仓库唯一
}
```

#### 5.3.1 认证中间件 (middleware/auth.ts)

```typescript
// backend/src/middleware/auth.ts
import { Request, Response, NextFunction } from 'express';
import { prisma } from '../config';
import bcrypt from 'bcrypt';

// 扩展Express Request类型
declare global {
  namespace Express {
    interface Request {
      user?: {
        id: number;
        username: string;
        realName: string;
        role: string;
      };
    }
  }
}

// Basic Auth 认证中间件
export async function authMiddleware(req: Request, res: Response, next: NextFunction) {
  try {
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith('Basic ')) {
      return res.status(401).json({
        code: 401,
        message: '未授权，请登录'
      });
    }
    
    // 解码 Basic Auth
    const base64Credentials = authHeader.split(' ')[1];
    const credentials = Buffer.from(base64Credentials, 'base64').toString('utf8');
    const [username, password] = credentials.split(':');
    
    // 查找用户
    const user = await prisma.user.findUnique({
      where: { username }
    });
    
    if (!user || user.status !== 1) {
      return res.status(401).json({
        code: 401,
        message: '用户名或密码错误'
      });
    }
    
    // 验证密码
    const isValid = await bcrypt.compare(password, user.password);
    
    if (!isValid) {
      return res.status(401).json({
        code: 401,
        message: '用户名或密码错误'
      });
    }
    
    // 注入用户信息到请求
    req.user = {
      id: user.id,
      username: user.username,
      realName: user.realName || '',
      role: user.role
    };
    
    next();
  } catch (error) {
    return res.status(500).json({
      code: 500,
      message: '认证失败'
    });
  }
}
```

```typescript
          skus: true

        take: pageSize
      }),

    ]);
    
    return {
      items: products,
      total,
      page,
      pageSize,

    };
  }

        status: data.status ?? 1
      }
    });

        brand: data.brand,

        price: data.price,

        mainImage: data.mainImage,
```

```typescript
// backend/src/services/sale.service.ts
import { prisma } from '../config';
import { generateCode } from '../utils/code';
import { inventoryService } from './inventory.service';

export class SaleService {
  
  // 创建销售订单
  async createOrder(data: {
    items: Array<{
      skuId: number;
      quantity: number;
      unitPrice: number;
    }>;
    memberId?: number;
    payWay: string;
    discountAmount?: number;
    remark?: string;
    cashierId: number;
  }) {
    // 计算订单金额
    let totalAmount = 0;
    let totalQuantity = 0;
    
    for (const item of data.items) {
      totalAmount += item.unitPrice * item.quantity;
      totalQuantity += item.quantity;
    }
    
    const discountAmount = data.discountAmount || 0;
    const payAmount = totalAmount - discountAmount;
    
    // 生成订单号
    const orderNo = await generateCode('XS');
    
    // 创建订单和订单明细
    const order = await prisma.$transaction(async (tx) => {
      // 创建订单
      const newOrder = await tx.order.create({
        data: {
          orderNo,
          memberId: data.memberId,
          cashierId: data.cashierId,
          totalQuantity,
          totalAmount,
          discountAmount,
          payAmount,
          payWay: JSON.stringify(data.payWay),
          status: 1
        }
      });
      
      // 创建订单明细
      for (const item of data.items) {
        await tx.orderItem.create({
          data: {
            orderId: newOrder.id,
            skuId: item.skuId,
            productName: '', // 可从SKU获取
            quantity: item.quantity,
            unitPrice: item.unitPrice,
            amount: item.unitPrice * item.quantity
          }
        });
        
        // 扣减库存
        await inventoryService.decreaseStock(tx, item.skuId, item.quantity);
      }
      
      // 计算并添加积分
      if (data.memberId) {
        const points = Math.floor(payAmount); // 1元=1积分
        await tx.member.update({
          where: { id: data.memberId },
          data: {
            points: { increment: points },
            totalConsume: { increment: payAmount }
          }
        });
        
        // 记录积分变动
        await tx.pointsLog.create({
          data: {
            memberId: data.memberId,
            changeType: 'earn',
            points,
            balance: 0, // 需要查询当前余额
            source: 'order',
            orderId: newOrder.id,
            description: `消费获得积分`
          }
        });
      }
      
      return newOrder;
    });
    
    return order;
  }
  
  // 获取订单列表
  async getOrders(params: {
    page: number;
    pageSize: number;
    startDate?: string;
    endDate?: string;
    status?: number;
  }) {
    const { page, pageSize, startDate, endDate, status } = params;
    
    const where: any = {};
    
    if (status !== undefined) {
      where.status = status;
    }
    
    if (startDate || endDate) {
      where.createdAt = {};
      if (startDate) {
        where.createdAt.gte = new Date(startDate);
      }
      if (endDate) {
        where.createdAt.lte = new Date(endDate + ' 23:59:59');
      }
    }
    
    const [orders, total] = await Promise.all([
      prisma.order.findMany({
        where,
        include: {
          member: true,
          cashier: { select: { id: true, username: true, realName: true } },
          items: { include: { sku: { include: { product: true } } } }
        },
        orderBy: { createdAt: 'desc' },
        skip: (page - 1) * pageSize,
        take: pageSize
      }),
      prisma.order.count({ where })
    ]);
    
    return {
      items: orders,
      total,
      page,
      pageSize,
      totalPages: Math.ceil(total / pageSize)
    };
  }
  
  // 获取订单详情
  async getOrderById(id: number) {
    return prisma.order.findUnique({
      where: { id },
      include: {
        member: true,
        cashier: { select: { id: true, username: true, realName: true } },
        items: { include: { sku: { include: { product: true } } } }
      }
    });
  }
  
  // 申请退货
  async createReturn(data: {
    orderId: number;
    items: Array<{ skuId: number; quantity: number }>;
    returnReason: string;
    operatorId: number;
  }) {
    const order = await prisma.order.findUnique({
      where: { id: data.orderId },
      include: { items: true }
    });
    
    if (!order) {
      throw new Error('订单不存在');
    }
    
    // 计算退货金额
    let returnAmount = 0;
    for (const item of data.items) {
      const orderItem = order.items.find(i => i.skuId === item.skuId);
      if (orderItem) {
        returnAmount += orderItem.unitPrice * item.quantity;
      }
    }
    
    // 生成退货单号
    const returnNo = await generateCode('TH');
    
    // 创建退货单
    const result = await prisma.$transaction(async (tx) => {
      // 创建退货单
      const returnOrder = await tx.orderReturn.create({
        data: {
          returnNo,
          orderId: data.orderId,
          returnReason: data.returnReason,
          returnAmount,
          status: 0,
          operatorId: data.operatorId
        }
      });
      
      // 退还库存
      for (const item of data.items) {
        await inventoryService.increaseStock(tx, item.skuId, item.quantity);
      }
      
      // 如果有会员，退还积分
      if (order.memberId) {
        const points = Math.floor(returnAmount);
        await tx.member.update({
          where: { id: order.memberId },
          data: {
            points: { decrement: points },
            totalConsume: { decrement: returnAmount }
          }
        });
      }
      
      return returnOrder;
    });
    
    return result;
  }
  
  // 日结
  async dailySettlement(data: { operatorId: number; settlementDate: string }) {
    const date = data.settlementDate || new Date().toISOString().split('T')[0];
    
    // 查询当日订单
    const startDate = new Date(date + ' 00:00:00');
    const endDate = new Date(date + ' 23:59:59');
    
    const orders = await prisma.order.findMany({
      where: {
        createdAt: { gte: startDate, lte: endDate },
        status: 1
      }
    });
    
    // 统计
    const salesCount = orders.length;
    const salesAmount = orders.reduce((sum, o) => sum + o.payAmount, 0);
    const discountAmount = orders.reduce((sum, o) => sum + o.discountAmount, 0);
    
    // 按支付方式统计
    let cashAmount = 0;
    let wechatAmount = 0;
    let alipayAmount = 0;
    
    for (const order of orders) {
      const payWay = JSON.parse(order.payWay || '{}');
      cashAmount += payWay.cash || 0;
      wechatAmount += payWay.wechat || 0;
      alipayAmount += payWay.alipay || 0;
    }
    
    // 查询退货
    const returns = await prisma.orderReturn.findMany({
      where: {
        createdAt: { gte: startDate, lte: endDate },
        status: { gte: 1 }
      }
    });
    
    const refundCount = returns.length;
    const refundAmount = returns.reduce((sum, r) => sum + r.returnAmount, 0);
    
    // 创建或更新日结记录
    const settlement = await prisma.dailySettlement.upsert({
      where: {
        settlementDate: new Date(date)
      },
      update: {
        salesCount,
        salesAmount,
        cashAmount,
        wechatAmount,
        alipayAmount,
        refundCount,
        refundAmount,
        discountAmount,
        operatorId: data.operatorId,
        status: 1
      },
      create: {
        settlementDate: new Date(date),
        salesCount,
        salesAmount,
        cashAmount,
        wechatAmount,
        alipayAmount,
        refundCount,
        refundAmount,
        discountAmount,
        operatorId: data.operatorId,
        status: 1
      }
    });
    
    return settlement;
  }
}

export const saleService = new SaleService();
```

```typescript
// backend/src/services/inventory.service.ts
import { Prisma } from '@prisma/client';

export class InventoryService {
  
  // 增加库存
  async increaseStock(tx: any, skuId: number, quantity: number) {
    const inventory = await tx.inventory.findUnique({
      where: { skuId_warehouseId: { skuId, warehouseId: 1 } }
    });
    
    if (inventory) {
      await tx.inventory.update({
        where: { id: inventory.id },
        data: {
          quantity: { increment: quantity },
          lastInTime: new Date()
        }
      });
    } else {
      await tx.inventory.create({
        data: {
          skuId,
          warehouseId: 1,
          quantity,
          lastInTime: new Date()
        }
      });
    }
    
    // 更新SKU库存
    await tx.sku.update({
      where: { id: skuId },
      data: { stockQuantity: { increment: quantity } }
    });
    
    // 记录库存变动
    await tx.inventoryLog.create({
      data: {
        skuId,
        changeType: 'in',
        changeQuantity: quantity,
        beforeQuantity: (inventory?.quantity || 0),
        afterQuantity: (inventory?.quantity || 0) + quantity,
        billType: 'order',
        operatorId: 1
      }
    });
  }
  
  // 减少库存
  async decreaseStock(tx: any, skuId: number, quantity: number) {
    const inventory = await tx.inventory.findUnique({
      where: { skuId_warehouseId: { skuId, warehouseId: 1 } }
    });
    
    if (!inventory || inventory.quantity < quantity) {
      throw new Error('库存不足');
    }
    
    await tx.inventory.update({
      where: { id: inventory.id },
      data: {
        quantity: { decrement: quantity },
        lastOutTime: new Date()
      }
    });
    
    // 更新SKU库存
    await tx.sku.update({
      where: { id: skuId },
      data: { stockQuantity: { decrement: quantity } }
    });
    
    // 记录库存变动
    await tx.inventoryLog.create({
      data: {
        skuId,
        changeType: 'out',
        changeQuantity: -quantity,
        beforeQuantity: inventory.quantity,
        afterQuantity: inventory.quantity - quantity,
        billType: 'order',
        operatorId: 1
      }
    });
  }
}
```

### 6.4 状态管理

```typescript
// frontend/src/stores/authStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { authApi } from '../api';

interface User {
  id: number;
  username: string;
  realName: string;
  role: string;
}

interface AuthState {
  user: User | null;
  isAuthenticated: boolean;
  login: (username: string, password: string) => Promise<void>;
  logout: () => void;
  checkAuth: () => Promise<void>;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      isAuthenticated: false,
      
      login: async (username: string, password: string) => {
        const credentials = btoa(`${username}:${password}`);
        localStorage.setItem('credentials', credentials);
        
        try {
          const user = await authApi.login(username, password);
          set({ user, isAuthenticated: true });
        } catch (error) {
          localStorage.removeItem('credentials');
          throw error;
        }
      },
      
      logout: () => {
        localStorage.removeItem('credentials');
        set({ user: null, isAuthenticated: false });
      },
      
      checkAuth: async () => {
        const credentials = localStorage.getItem('credentials');
        if (!credentials) {
          set({ isAuthenticated: false });
          return;
        }
        
        try {
          const user = await authApi.getCurrentUser();
          set({ user, isAuthenticated: true });
        } catch {
          set({ user: null, isAuthenticated: false });
        }
      }
    }),
    {
      name: 'auth-storage'
    }
  )
);
```
